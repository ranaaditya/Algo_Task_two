#include<bits/stdc++.h>
#define ll long long  
using namespace std;

stack<ll> sortStack(stack<ll> &input) 
{ 
    stack<ll> tmpStack; 
  
    while (!input.empty()) 
    { 
         
        ll tmp = input.top(); 
        input.pop(); 
  
        
        while (!tmpStack.empty() && tmpStack.top() < tmp) 
        { 
           
            input.push(tmpStack.top()); 
            tmpStack.pop(); 
        } 
  
       
        tmpStack.push(tmp); 
    } 
  
    return tmpStack; 
}
